社員マスタ																									
CREATE TABLE `member` (																									
`id` 社員番号 int(11) NOT NULL,																									
`name` 氏名 varchar(45) DEFAULT NULL,																									
`initial` イニシャル氏名 varchar(45) DEFAULT NULL,																									
`adress` 住所 varchar(45) DEFAULT NULL,																									
`station` 最寄駅 varchar(45) DEFAULT NULL,																									
`information` 給与情報 varchar(45) DEFAULT NULL,																									
`attach` 所属 varchar(45) DEFAULT NULL,																									
`position` 役職 varchar(45) DEFAULT NULL,																									
`status` ステータス(提案可否,待機,終業) varchar(110) NOT NULL,																									
PRIMARY KEY (`id`)																									
) ENGINE=InnoDB DEFAULT CHARSET=utf8																								
																									
給与情報																									
CREATE TABLE `money` (																									
`id` 社員番号 int(11) NOT NULL,																									
`salary` 基本給 varchar(45) DEFAULT NULL,																									
`pay` 業務手当 varchar(45) DEFAULT NULL,																									
`product` 成果給 varchar(45) DEFAULT NULL,																									
`insurance` 雇用保険 varchar(45) DEFAULT NULL,																									
`expenses` 交通費 varchar(45) DEFAULT NULL,																									
PRIMARY KEY (`id`)																									
) ENGINE=InnoDB DEFAULT CHARSET=utf8																									
																									
提案状況																									
CREATE TABLE `overture` (																									
`id` 引き合い番号 int(11) NOT NULL,																									
`numero` 社員番号 varchar(45) DEFAULT NULL,																									
`state` 状況 (提案,面談1...)varchar(110),																									
PRIMARY KEY (`id`)																									
) ENGINE=InnoDB DEFAULT CHARSET=utf8																									
																									
スキル情報																									
CREATE TABLE `skill` (

`skills` 基本スキル情報 varchar(45) DEFAULT NULL,																									
`comment` コメント varchar(45) DEFAULT NULL,

) ENGINE=InnoDB DEFAULT CHARSET=utf8																									
																									
																									
案件マスタ																									
CREATE TABLE `deal` (
`id_2` 契約書番号 int(11) NOT NULL,																									
`id` 管理担当 (社員番号) int(11) NOT NULL,																									
`numero` 取引先 varchar(45) DEFAULT NULL,																									
`dealcol` 勤務先 varchar(45) DEFAULT NULL,																									
`customer` 社員番号 varchar(45) DEFAULT NULL,																									
`work` ステータス varchar(45) DEFAULT NULL,																									
`cost` 単金 (就業,短期..) varchar(45) DEFAULT NULL,																									
`require` 条件 varchar(45) DEFAULT NULL,																									
`term`  期日(開始,終了) varchar(45) DEFAULT NULL,																									
PRIMARY KEY (`id`)																									
) ENGINE=InnoDB DEFAULT CHARSET=utf8
